title: capitalize
description: Capitalize the first letter of a string while keeping the rest unchanged.
---

import { Callout } from "fumadocs-ui/components/callout";

## Installation

```bash
npx react-usekit@latest init # Select capitalize from the utilities list
```

## Usage

```tsx
import { capitalize } from "./utils/capitalize";

function App() {
  const text = "hello world";
  const capitalizedText = capitalize(text);

  return (
    <div>
      <p>Original: {text}</p>
      <p>Capitalized: {capitalizedText}</p>
    </div>
  );
}
```

## API

### Parameters

| Parameter | Type     | Description              |
| --------- | -------- | ------------------------ |
| `str`     | `string` | The string to capitalize |

### Returns

| Type     | Description                                  |
| -------- | -------------------------------------------- |
| `string` | The string with the first letter capitalized |

## Examples

### Basic Usage

```tsx
import { capitalize } from "./utils/capitalize";

console.log(capitalize("hello")); // "Hello"
console.log(capitalize("world")); // "World"
console.log(capitalize("JAVASCRIPT")); // "JAVASCRIPT"
console.log(capitalize("react")); // "React"
```

### User Name Display

```tsx
import { capitalize } from "./utils/capitalize";

function UserProfile({ user }) {
  return (
    <div>
      <h2>
        Welcome, {capitalize(user.firstName)} {capitalize(user.lastName)}!
      </h2>
    </div>
  );
}
```

### Form Field Labels

```tsx
import { capitalize } from "./utils/capitalize";

function DynamicForm({ fields }) {
  return (
    <form>
      {fields.map((field) => (
        <div key={field.name}>
          <label>{capitalize(field.name)}</label>
          <input type={field.type} name={field.name} />
        </div>
      ))}
    </form>
  );
}
```

### Navigation Menu

```tsx
import { capitalize } from "./utils/capitalize";

function Navigation({ menuItems }) {
  return (
    <nav>
      {menuItems.map((item) => (
        <a key={item.id} href={item.href}>
          {capitalize(item.name)}
        </a>
      ))}
    </nav>
  );
}
```

### Error Messages

```tsx
import { capitalize } from "./utils/capitalize";

function ErrorMessage({ field, error }) {
  return (
    <div className="error">
      {capitalize(field)} {error}
    </div>
  );
}

// Usage
<ErrorMessage field="email" error="is required" />;
// Renders: "Email is required"
```
