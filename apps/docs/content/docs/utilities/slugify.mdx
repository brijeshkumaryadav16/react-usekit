# slugify

The `slugify` utility function converts text into URL-friendly slugs by removing special characters, converting to lowercase, and replacing spaces with hyphens. It's perfect for creating clean URLs, file names, and identifiers.

## Usage

```javascript
import { slugify } from "./utils/slugify";

const slug = slugify("Hello World!");
console.log(slug); // "hello-world"

const blogSlug = slugify("My First Blog Post: Getting Started with React");
console.log(blogSlug); // "my-first-blog-post-getting-started-with-react"
```

## API Reference

### Parameters

- `text` (string): The text to convert into a slug

### Returns

- `string`: The URL-friendly slug

### Blog System

```javascript
function BlogPostEditor() {
  const [title, setTitle] = useState("");
  const [content, setContent] = useState("");
  const [slug, setSlug] = useState("");
  const [isCustomSlug, setIsCustomSlug] = useState(false);

  // Auto-generate slug from title
  useEffect(() => {
    if (!isCustomSlug && title) {
      setSlug(slugify(title));
    }
  }, [title, isCustomSlug]);

  const handleSlugChange = (e) => {
    setSlug(slugify(e.target.value));
    setIsCustomSlug(true);
  };

  const handleSubmit = async (e) => {
    e.preventDefault();

    const post = {
      title,
      content,
      slug: slug || slugify(title),
      publishedAt: new Date().toISOString(),
    };

    await savePost(post);
  };

  return (
    <form onSubmit={handleSubmit} className="blog-editor">
      <div className="form-group">
        <label>Title</label>
        <input
          value={title}
          onChange={(e) => setTitle(e.target.value)}
          placeholder="Enter blog post title"
        />
      </div>

      <div className="form-group">
        <label>URL Slug</label>
        <div className="slug-input">
          <span className="base-url">https://myblog.com/posts/</span>
          <input
            value={slug}
            onChange={handleSlugChange}
            placeholder="auto-generated-from-title"
          />
        </div>
        <small>URL-friendly version of your title</small>
      </div>

      <div className="form-group">
        <label>Content</label>
        <textarea
          value={content}
          onChange={(e) => setContent(e.target.value)}
          rows={10}
          placeholder="Write your blog post content"
        />
      </div>

      <button type="submit">Publish Post</button>
    </form>
  );
}
```

### File Management System

```javascript
function FileUploader() {
  const [files, setFiles] = useState([]);

  const handleFileUpload = (uploadedFiles) => {
    const processedFiles = Array.from(uploadedFiles).map((file) => {
      const nameWithoutExt = file.name.replace(/\.[^/.]+$/, "");
      const extension = file.name.split(".").pop();
      const slugifiedName = slugify(nameWithoutExt);

      return {
        original: file,
        name: file.name,
        slugifiedName: `${slugifiedName}.${extension}`,
        size: file.size,
        type: file.type,
        id: generateId(),
      };
    });

    setFiles((prev) => [...prev, ...processedFiles]);
  };

  const saveFiles = async () => {
    for (const file of files) {
      const formData = new FormData();
      formData.append("file", file.original);
      formData.append("filename", file.slugifiedName);

      await fetch("/api/upload", {
        method: "POST",
        body: formData,
      });
    }
  };

  return (
    <div className="file-uploader">
      <input
        type="file"
        multiple
        onChange={(e) => handleFileUpload(e.target.files)}
      />

      {files.length > 0 && (
        <div className="file-list">
          <h3>Files to Upload</h3>
          {files.map((file) => (
            <div key={file.id} className="file-item">
              <div className="file-info">
                <strong>Original:</strong> {file.name}
                <br />
                <strong>Saved as:</strong> {file.slugifiedName}
              </div>
            </div>
          ))}
          <button onClick={saveFiles}>Upload All Files</button>
        </div>
      )}
    </div>
  );
}
```

```javascript
function ProductManager() {

      ...productData,
```

### SEO URL Generator

```javascript
function SEOUrlGenerator() {
  const [pages, setPages] = useState([]);

  const generateSEOUrl = (title, category, tags) => {
    const baseSlug = slugify(title);
    const categorySlug = slugify(category);
    const tagSlugs = tags.map((tag) => slugify(tag));

    return {
      basic: `/${baseSlug}`,
      withCategory: `/${categorySlug}/${baseSlug}`,
      withTags: `/${baseSlug}?tags=${tagSlugs.join(",")}`,
      full: `/${categorySlug}/${baseSlug}?tags=${tagSlugs.join(",")}`,
    };
  };

  const addPage = (pageData) => {
    const urls = generateSEOUrl(
      pageData.title,
      pageData.category,
      pageData.tags
    );

    setPages((prev) => [
      ...prev,
      {
        ...pageData,
        id: generateId(),
        urls,
        selectedUrl: urls.withCategory,
      },
    ]);
  };

  return (
    <div className="seo-url-generator">
      <PageForm onSubmit={addPage} />

      <div className="generated-urls">
        {pages.map((page) => (
          <div key={page.id} className="url-options">
            <h3>{page.title}</h3>
            <div className="url-variants">
              <label>
                <input type="radio" name={`url-${page.id}`} value="basic" />
                Basic: {page.urls.basic}
              </label>
              <label>
                <input
                  type="radio"
                  name={`url-${page.id}`}
                  value="withCategory"
                  defaultChecked
                />
                With Category: {page.urls.withCategory}
              </label>
              <label>
                <input type="radio" name={`url-${page.id}`} value="full" />
                Full: {page.urls.full}
              </label>
            </div>
          </div>
        ))}
      </div>
    </div>
  );
}
```

### Breadcrumb Generator

```javascript
function BreadcrumbGenerator({ pathname }) {
  const generateBreadcrumbs = (path) => {
    const segments = path.split("/").filter(Boolean);

    return segments.map((segment, index) => {
      const decodedSegment = decodeURIComponent(segment);
      // Convert slug back to readable text
      const readable = decodedSegment
        .split("-")
        .map((word) => word.charAt(0).toUpperCase() + word.slice(1))
        .join(" ");

      const href = "/" + segments.slice(0, index + 1).join("/");

      return {
        label: readable,
        href,
        isLast: index === segments.length - 1,
      };
    });
  };

  const breadcrumbs = [
    { label: "Home", href: "/", isLast: false },
    ...generateBreadcrumbs(pathname),
  ];

  return (
    <nav className="breadcrumbs">
      {breadcrumbs.map((crumb, index) => (
        <span key={index}>
          {index > 0 && <span className="separator"> / </span>}
          {crumb.isLast ? (
            <span className="current">{crumb.label}</span>
          ) : (
            <a href={crumb.href}>{crumb.label}</a>
          )}
        </span>
      ))}
    </nav>
  );
}

// /blog/my-first-post → Home / Blog / My First Post
// /products/electronics/smartphones → Home / Products / Electronics / Smartphones
```

## Notes

- Converts text to lowercase and replaces spaces with hyphens
- Removes special characters and accents for clean URLs
- Handles multiple spaces and consecutive hyphens
- Perfect for creating SEO-friendly URLs and file names
- Consistent output makes URLs predictable and clean
- Consider length limits for very long titles (usually 50-60 characters)
