```package-install
shadcn@latest add "https://react-usekit.vercel.app/r/is-valid-url.json"
```

## Usage

```tsx
import { isURLValid } from "@/lib/is-valid-url";

function App() {
  const validURL = "https://example.com";
  const invalidURL = "not-a-url";

  return (
    <div>
      <p>
        {validURL}: {isURLValid(validURL) ? "Valid" : "Invalid"}
      </p>
      <p>
        {invalidURL}: {isURLValid(invalidURL) ? "Valid" : "Invalid"}
      </p>
    </div>
  );
}
```

## API

### Parameters

| Parameter | Type     | Default | Description                |
| --------- | -------- | ------- | -------------------------- |
| `url`     | `string` | -       | The URL string to validate |
| `options` | `object` | `{}`    | Configuration options      |

### Returns

| Type      | Description                               |
| --------- | ----------------------------------------- |
| `boolean` | True if the URL is valid, false otherwise |
